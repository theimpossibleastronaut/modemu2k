582/* ── Steppable event-loop API ───────────────────────────────────────
583 *
584 * The functions below let a host application integrate modemu2k into
585 * its own event loop (poll/epoll/select/etc.) instead of handing the
586 * fds to the blocking m2k_run() above.
587 *
588 * Usage:
589 * while (!m2k_run_done(ctx)) {
590 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
591 * size_t nfds = M2K_MAX_POLLFDS;
592 * int timeout_ms;
593 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
594 * // Splice fds into your own pollset, call poll()/epoll_wait().
595 * // On return, write back the .revents the OS reported into the
596 * // corresponding entries of `fds`.
597 * m2k_step(ctx, fds, nfds);
598 * }
599 *
600 * Dialing (ATD) is non-blocking: m2k_step() transitions the context
601 * into an internal DIAL state on ATD, returns control immediately,
602 * and resumes the connect() across subsequent m2k_step() calls. The
603 * host event loop is not held up by connect(). m2k_get_pollfds()
604 * publishes the in-progress socket fd while DIAL is active, so the
605 * host pollset stays accurate without special-casing.
606 *
607 * Answering (ATA, or S0 auto-answer) works the same way: the context
608 * parks in an internal ANSWER state while waiting for a caller within
609 * the S7 window, and m2k_get_pollfds() publishes the answer listener
610 * fd for the duration.
611 */
612
614#define M2K_MAX_POLLFDS 3
