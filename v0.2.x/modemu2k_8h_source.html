541/* ── Steppable event-loop API ───────────────────────────────────────
542 *
543 * The functions below let a host application integrate modemu2k into
544 * its own event loop (poll/epoll/select/etc.) instead of handing the
545 * fds to the blocking m2k_run() above.
546 *
547 * Usage:
548 * while (!m2k_run_done(ctx)) {
549 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
550 * size_t nfds = M2K_MAX_POLLFDS;
551 * int timeout_ms;
552 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
553 * // Splice fds into your own pollset, call poll()/epoll_wait().
554 * // On return, write back the .revents the OS reported into the
555 * // corresponding entries of `fds`.
556 * m2k_step(ctx, fds, nfds);
557 * }
558 *
559 * Dialing (ATD) is non-blocking: m2k_step() transitions the context
560 * into an internal DIAL state on ATD, returns control immediately,
561 * and resumes the connect() across subsequent m2k_step() calls. The
562 * host event loop is not held up by connect(). m2k_get_pollfds()
563 * publishes the in-progress socket fd while DIAL is active, so the
564 * host pollset stays accurate without special-casing.
565 */
566
568#define M2K_MAX_POLLFDS 3
