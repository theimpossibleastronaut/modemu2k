242/* requirements for a telnet option */
243typedef enum
244{
245 TOR_MUSTNOT, /* disable the opt or disconnect */
246 TOR_BETTERNOT, /* disable the opt or enable unwillingly */
247 TOR_NEUTRAL, /* modemu doesn't initiate any action for the opt.
248 {en,dis}able the opt as the peer requests. */
249 TOR_BETTER, /* enable the opt or disable unwillingly */
250 TOR_MUST /* enable the opt or disconnect */
251} TelOptReq;
252
253typedef struct
254{
255 TelOptReq req; /* requirement for the opt */
256 int state; /* current state (enabled:1 or disabled:0) */
257 int pending; /* state is pending (requested but no reply yet) */
258} TelOptState;
259
260typedef struct
261{
262 int opt; /* an telnet option. TELOPT_XXX */
263 TelOptState local; /* local state of the option */
264 TelOptState remote; /* remote state of the option */
265} TelOptStates;
266

268{
269 int binsend; /* local binary opt is enabled */
270 int binrecv; /* remote binary opt is enabled */
271 int sgasend; /* local SGA opt is enabled (char-at-a-time mode) */
272 int sentReqs; /* have sent option requests to the peer
273 or skip sending them */
274 TelOptStates **stTab; /* = stTab[] in telopt.c */
275};
