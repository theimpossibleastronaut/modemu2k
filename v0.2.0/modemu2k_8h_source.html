323/* ── Steppable event-loop API ───────────────────────────────────────
324 *
325 * The functions below let a host application integrate modemu2k into
326 * its own event loop (poll/epoll/select/etc.) instead of handing the
327 * fds to the blocking m2k_run() above.
328 *
329 * Usage:
330 * while (!m2k_run_done(ctx)) {
331 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
332 * size_t nfds = M2K_MAX_POLLFDS;
333 * int timeout_ms;
334 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
335 * // Splice fds into your own pollset, call poll()/epoll_wait().
336 * // On return, write back the .revents the OS reported into the
337 * // corresponding entries of `fds`.
338 * m2k_step(ctx, fds, nfds);
339 * }
340 *
341 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
342 * the connect() call will block your event loop briefly. That will be
343 * addressed in a later, non-blocking-dial pass.
344 */
345
347#define M2K_MAX_POLLFDS 3
