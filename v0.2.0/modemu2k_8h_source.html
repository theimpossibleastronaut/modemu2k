359/* ── Steppable event-loop API ───────────────────────────────────────
360 *
361 * The functions below let a host application integrate modemu2k into
362 * its own event loop (poll/epoll/select/etc.) instead of handing the
363 * fds to the blocking m2k_run() above.
364 *
365 * Usage:
366 * while (!m2k_run_done(ctx)) {
367 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
368 * size_t nfds = M2K_MAX_POLLFDS;
369 * int timeout_ms;
370 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
371 * // Splice fds into your own pollset, call poll()/epoll_wait().
372 * // On return, write back the .revents the OS reported into the
373 * // corresponding entries of `fds`.
374 * m2k_step(ctx, fds, nfds);
375 * }
376 *
377 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
378 * the connect() call will block your event loop briefly. That will be
379 * addressed in a later, non-blocking-dial pass.
380 */
381
383#define M2K_MAX_POLLFDS 3
