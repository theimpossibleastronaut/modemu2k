261/* ── Steppable event-loop API ───────────────────────────────────────
262 *
263 * The functions below let a host application integrate modemu2k into
264 * its own event loop (poll/epoll/select/etc.) instead of handing the
265 * fds to the blocking m2k_run() above.
266 *
267 * Usage:
268 * while (!m2k_run_done(ctx)) {
269 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
270 * size_t nfds = M2K_MAX_POLLFDS;
271 * int timeout_ms;
272 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
273 * // Splice fds into your own pollset, call poll()/epoll_wait().
274 * // On return, write back the .revents the OS reported into the
275 * // corresponding entries of `fds`.
276 * m2k_step(ctx, fds, nfds);
277 * }
278 *
279 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
280 * the connect() call will block your event loop briefly. That will be
281 * addressed in a later, non-blocking-dial pass.
282 */
