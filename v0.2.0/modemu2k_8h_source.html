400/* ── Steppable event-loop API ───────────────────────────────────────
401 *
402 * The functions below let a host application integrate modemu2k into
403 * its own event loop (poll/epoll/select/etc.) instead of handing the
404 * fds to the blocking m2k_run() above.
405 *
406 * Usage:
407 * while (!m2k_run_done(ctx)) {
408 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
409 * size_t nfds = M2K_MAX_POLLFDS;
410 * int timeout_ms;
411 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
412 * // Splice fds into your own pollset, call poll()/epoll_wait().
413 * // On return, write back the .revents the OS reported into the
414 * // corresponding entries of `fds`.
415 * m2k_step(ctx, fds, nfds);
416 * }
417 *
418 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
419 * the connect() call will block your event loop briefly. That will be
420 * addressed in a later, non-blocking-dial pass.
421 */
422
424#define M2K_MAX_POLLFDS 3
