320/* ── Steppable event-loop API ───────────────────────────────────────
321 *
322 * The functions below let a host application integrate modemu2k into
323 * its own event loop (poll/epoll/select/etc.) instead of handing the
324 * fds to the blocking m2k_run() above.
325 *
326 * Usage:
327 * while (!m2k_run_done(ctx)) {
328 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
329 * size_t nfds = M2K_MAX_POLLFDS;
330 * int timeout_ms;
331 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
332 * // Splice fds into your own pollset, call poll()/epoll_wait().
333 * // On return, write back the .revents the OS reported into the
334 * // corresponding entries of `fds`.
335 * m2k_step(ctx, fds, nfds);
336 * }
337 *
338 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
339 * the connect() call will block your event loop briefly. That will be
340 * addressed in a later, non-blocking-dial pass.
341 */
342
344#define M2K_MAX_POLLFDS 3
