445/* ── Steppable event-loop API ───────────────────────────────────────
446 *
447 * The functions below let a host application integrate modemu2k into
448 * its own event loop (poll/epoll/select/etc.) instead of handing the
449 * fds to the blocking m2k_run() above.
450 *
451 * Usage:
452 * while (!m2k_run_done(ctx)) {
453 * struct pollfd fds[M2K_MAX_POLLFDS]; // caller-provided buffer
454 * size_t nfds = M2K_MAX_POLLFDS;
455 * int timeout_ms;
456 * m2k_get_pollfds(ctx, fds, &nfds, &timeout_ms);
457 * // Splice fds into your own pollset, call poll()/epoll_wait().
458 * // On return, write back the .revents the OS reported into the
459 * // corresponding entries of `fds`.
460 * m2k_step(ctx, fds, nfds);
461 * }
462 *
463 * Dialing (ATD) is still synchronous inside m2k_step() in this phase —
464 * the connect() call will block your event loop briefly. That will be
465 * addressed in a later, non-blocking-dial pass.
466 */
467
469#define M2K_MAX_POLLFDS 3
